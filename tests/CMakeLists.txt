cmake_minimum_required(VERSION 3.21)
project(RingByteBufferTests)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)
set(CMAKE_CXX_EXTENSIONS FALSE)

add_executable(rbb_validate_automatic rbb_validate_automatic.cpp)
target_link_libraries(rbb_validate_automatic PRIVATE RingByteBuffer)
target_compile_options(rbb_validate_automatic PRIVATE ${rbb_compile_options})

add_executable(rbb_validate_handwritten rbb_validate_handwritten.cpp)
target_link_libraries(rbb_validate_handwritten PRIVATE RingByteBuffer)
target_compile_options(rbb_validate_handwritten PRIVATE ${rbb_compile_options})
